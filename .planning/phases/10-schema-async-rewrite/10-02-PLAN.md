PostgreSQL Drizzle query conventions:
- `.get()` -> `await ...then(r => r[0] ?? null)` OR `const [row] = await ...`
- `.all()` -> `await ...` (already returns array)
- `.run()` -> `await ...` (just await, no return value needed)
- `.returning().get()` -> `const [row] = await ...returning()`

Apply these mechanical transformations to every DB call site in all 6 files. The conversion rules are:

**Rule 1: `.get()` -> await + array index**
```typescript
// Before
const row = db.select().from(table).where(condition).get();
// After
const row = (await db.select().from(table).where(condition))[0] ?? null;
```

**Rule 2: `.all()` -> await**
```typescript
// Before
const rows = db.select().from(table).all();
// After
const rows = await db.select().from(table);
```

**Rule 3: `.run()` -> await**
```typescript
// Before
db.insert(table).values(data).run();
// After
await db.insert(table).values(data);
```

**Rule 4: `.returning().get()` -> await + destructure**
```typescript
// Before
const script = db.insert(scripts).values({...}).returning().get();
// After
const [script] = await db.insert(scripts).values({...}).returning();
```

**Rule 7: getLastSyncTime epoch fix (CRITICAL — Pitfall 3 from research)**
In `src/app/actions/metrics.ts`, function `getLastSyncTime()` around line 223:
```typescript
// Before
const result = db
  .select({ maxSync: sql<number>`MAX(${videoMetrics.lastSyncedAt})` })
  .from(videoMetrics)
  .get();
if (!result?.maxSync) return null;
return new Date(result.maxSync * 1000);

// After
const result = (await db
  .select({ maxSync: sql<Date>`MAX(${videoMetrics.lastSyncedAt})` })
  .from(videoMetrics))[0] ?? null;
if (!result?.maxSync) return null;
return result.maxSync;
```
Changes: `sql<number>` -> `sql<Date>`, add `await` + `[0]`, remove `* 1000` multiplication, return `result.maxSync` directly (PostgreSQL timestamp already returns Date).

**File contents:**

```typescript
import { getDb } from "../src/lib/db";
import { scripts, beats, videos, videoMetrics } from "../src/lib/db/schema";
import { eq } from "drizzle-orm";

async function main() {
  const db = getDb();
  console.log("Smoke test: connecting to Supabase...\n");

  // Test 1: SELECT from each table (verifies connection + schema)
  console.log("1. Testing SELECT on all 4 tables...");
  const scriptRows = await db.select().from(scripts);
  console.log(`   scripts: ${scriptRows.length} rows`);
  const beatRows = await db.select().from(beats);
  console.log(`   beats: ${beatRows.length} rows`);
  const videoRows = await db.select().from(videos);
  console.log(`   videos: ${videoRows.length} rows`);
  const metricRows = await db.select().from(videoMetrics);
  console.log(`   video_metrics: ${metricRows.length} rows`);

  // Test 2: INSERT + SELECT + DELETE round-trip
  console.log("\n2. Testing INSERT -> SELECT -> DELETE round-trip...");
  const [inserted] = await db.insert(scripts).values({
    title: "__smoke_test__",
    format: "the-bug",
    status: "draft",
  }).returning();
  console.log(`   Inserted script id=${inserted.id}, title="${inserted.title}"`);

  const fetched = (await db.select().from(scripts).where(eq(scripts.id, inserted.id)))[0] ?? null;
  if (!fetched || fetched.title !== "__smoke_test__") {
    throw new Error("SELECT after INSERT returned wrong data");
  }
  console.log(`   Fetched back: id=${fetched.id}, title="${fetched.title}"`);

  await db.delete(scripts).where(eq(scripts.id, inserted.id));
  const afterDelete = (await db.select().from(scripts).where(eq(scripts.id, inserted.id)))[0] ?? null;
  if (afterDelete) {
    throw new Error("Row still exists after DELETE");
  }
  console.log("   Deleted successfully, verified gone.");

  // Test 3: Verify timestamp columns return Date objects
  console.log("\n3. Testing timestamp type...");
  const [tsTest] = await db.insert(scripts).values({
    title: "__ts_test__",
    format: "the-bug",
    status: "draft",
  }).returning();
  if (!(tsTest.createdAt instanceof Date)) {
    throw new Error(`createdAt is ${typeof tsTest.createdAt}, expected Date`);
  }
  console.log(`   createdAt is Date: ${tsTest.createdAt.toISOString()}`);
  await db.delete(scripts).where(eq(scripts.id, tsTest.id));

  console.log("\n--- ALL SMOKE TESTS PASSED ---");
  process.exit(0);
}

main().catch((err) => {
  console.error("\n--- SMOKE TEST FAILED ---");
  console.error(err);
  process.exit(1);
});
```
